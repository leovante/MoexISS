docker rm -f $(docker ps -aq | awk '{print $1}')
# Nukes images:
docker rmi -f $(docker images | awk '{print $3}')
# Nukes volumes:
docker-compose down -v
# Re-Builds docker-compose:
docker-compose build
# Stand everything back up:
docker-compose up
